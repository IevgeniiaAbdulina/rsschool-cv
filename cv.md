```js
    function palindrome(str) {
        var regex = /[a-z]|[0-9]+/g;
        var expression = str.toLowerCase().split("").map(elem => elem.match(regex)).join("");
        var reverse = expression.split("").reverse().join("");
        return expression === reverse;
    }

    palindrome("eye");
```
